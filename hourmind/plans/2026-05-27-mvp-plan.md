# HourMind MVP 实现计划

> **目标：** 构建最小可用版本 —— 认证登录 + API Key 管理 + 基础智能对话
> **技术栈：** Node.js + Express + Prisma + SQLite + TypeScript / Vue 3 + Vite + Pinia + Element Plus
> **核心理念：** 每个文件独立、每行代码都有注释、每个变量都能看懂

---

- [ ] **Step 4：创建数据库 Schema**

创建文件 `hourmind/server/prisma/schema.prisma`：

```prisma
// ============================================================
// Prisma Schema —— 定义数据库所有表的结构
// 每次修改这个文件后，需要执行 npm run db:migrate
// SQLite 不支持 enum 类型，所以我们用 String 代替
// ============================================================

// 告诉 Prisma 生成 JavaScript 客户端代码
generator client {
  provider = "prisma-client-js"
}

// 数据库连接配置
// SQLite 是一个文件型数据库，数据存在 db/hourmind.db 这个文件里
datasource db {
  provider = "sqlite"
  url      = "file:./db/hourmind.db"
}

// ============================================================
// 系统配置表 —— 存密码哈希、用户设置等
// 用 key-value 结构，一个 key 一行数据
// ============================================================
model AppConfig {
  key   String @id    // 配置项名称，主键
  value String        // 配置项的值
}

// ============================================================
// AI 厂商表 —— 存储支持的 AI 服务商信息（如 OpenAI、Claude）
// ============================================================
model AiProvider {
  id        String   @id @default(uuid())   // 自动生成 UUID 作为主键
  name      String                          // 显示名称，如 "OpenAI"
  slug      String   @unique               // 唯一标识，如 "openai"，不能重复
  baseUrl   String                         // API 接口地址
  logoUrl   String?                        // Logo 图片地址，可选
  isActive  Boolean  @default(true)        // 是否启用，默认 true
  createdAt DateTime @default(now())       // 创建时间，自动填当前时间

  apiKeys   ApiKey[]  // 一个厂商有多个 Key（一对多关系）
}

// ============================================================
// API Key 表 —— 存储用户添加的 AI 服务商密钥
// ============================================================
model ApiKey {
  id           String    @id @default(uuid())
  providerId   String                      // 外键，属于哪个厂商
  provider     AiProvider @relation(fields: [providerId], references: [id])
  alias        String                      // 别名，方便用户识别（如 "主力Key"）
  encryptedKey String                      // AES 加密后的 Key 值
  keySuffix    String                      // Key 最后 6 位明文（列表里识别用）
  tags         String    @default("[]")    // 标签，JSON 数组字符串
  status       String    @default("active") // active / disabled / error
  isDeleted    Boolean   @default(false)   // 软删除标记
  deletedAt    DateTime?                   // 什么时候删除的
  createdAt    DateTime  @default(now())
  updatedAt    DateTime  @updatedAt        // 修改时自动更新时间

  testLogs      KeyTestLog[]       // 一个 Key 有多次测试记录
  usageRecords  TokenUsageRecord[] // 一个 Key 有多次用量记录
}

// ============================================================
// Key 测试日志表 —— 记录每次测试 Key 连通性的结果
// ============================================================
model KeyTestLog {
  id           String   @id @default(uuid())
  keyId        String
  key          ApiKey   @relation(fields: [keyId], references: [id])
  isSuccess    Boolean                     // 测试是否成功
  latencyMs    Int                         // 响应延迟（毫秒）
  errorMessage String?                     // 失败时的错误信息
  testedAt     DateTime @default(now())
}

// ============================================================
// Token 用量记录表 —— 记录每次 AI 调用的 Token 消耗
// ============================================================
model TokenUsageRecord {
  id                 String   @id @default(uuid())
  keyId              String
  key                ApiKey   @relation(fields: [keyId], references: [id])
  modelName          String                  // 使用的模型名称
  promptTokens       Int                     // 输入消耗的 Token 数
  completionTokens   Int                     // 输出消耗的 Token 数
  estimatedCostCents Int     @default(0)     // 预估费用（美元分）
  recordedAt         DateTime @default(now())
  sessionId          String?                 // 关联的对话 ID
}

// ============================================================
// 会话表 —— 一次对话就是一个会话
// ============================================================
model Conversation {
  id           String     @id @default(uuid())
  title        String                      // 会话标题
  model        String                      // 使用的模型名
  status       String     @default("active") // active / archived
  isPinned     Boolean    @default(false)  // 是否置顶
  isStarred    Boolean    @default(false)  // 是否星标
  totalTokens  Int        @default(0)      // 累计 Token 消耗
  messageCount Int        @default(0)      // 消息总数
  createdAt    DateTime   @default(now())
  updatedAt    DateTime   @updatedAt

  messages Message[]  // 一个会话有多条消息
}

// ============================================================
// 消息表 —— 会话中的每一条消息
// ============================================================
model Message {
  id             String       @id @default(uuid())
  conversationId String
  conversation   Conversation @relation(fields: [conversationId], references: [id])
  role           String                      // user / assistant / system
  content        String                      // 消息内容
  model          String?                     // 本条回复使用的模型
  tokenCount     Int?                        // Token 消耗
  metadata       String      @default("{}")  // 扩展字段，JSON 字符串
  createdAt      DateTime    @default(now())
}
```

```typescript
// ============================================================
// db.ts —— 数据库连接模块
// Prisma Client 是数据库操作的入口，所有对数据库的增删改查
// 都通过 prisma 这个对象来进行
// 这里确保全局只有一个 Prisma Client 实例（单例模式）
// ============================================================

// 导入 Prisma Client 类
import { PrismaClient } from '@prisma/client'

// 声明全局变量类型（TypeScript 需要）
// globalThis 是 Node.js 的全局对象
const globalForPrisma = globalThis as unknown as {
  prisma: PrismaClient | undefined
}

// 如果全局中已有实例就复用，没有就新建
// 这样在开发时热重载不会创建多个数据库连接
export const prisma = globalForPrisma.prisma ?? new PrismaClient()

// 把实例保存到全局，下次热重载就能复用了
// 注意：生产环境不要这样做，这里仅为了方便开发
if (process.env.NODE_ENV !== 'production') {
  globalForPrisma.prisma = prisma
}
```

- [ ] **Step 6：创建启动入口文件**

创建文件 `hourmind/server/src/index.ts`：

```typescript
// ============================================================
// index.ts —— 后端服务的启动入口文件
// 它负责：
//   1. 加载环境变量
//   2. 启动 Express HTTP 服务
//   3. 启动 WebSocket 服务
// 通常用 node src/index.ts 或 npm run dev 启动
// ============================================================

// 加载 .env 文件中的环境变量到 process.env
import dotenv from 'dotenv'
dotenv.config()

// 导入 Express —— Node.js 最流行的 Web 框架
import express from 'express'
// 导入 cors —— 允许前端跨域访问
import cors from 'cors'
// 导入 Node.js 内置的 http 模块（创建底层 HTTP 服务器）
import http from 'http'

// 导入我们的 WebSocket 服务
import { createWsServer } from './wsServer'

// ==================== 1. 创建 Express 应用 ====================

const app = express()

// 允许来自前端地址的跨域请求
// 开发时前端跑在 5173 端口，后端在 3000，不同端口就是"跨域"
app.use(cors({
  origin: ['http://localhost:5173', 'http://127.0.0.1:5173'],
  credentials: true  // 允许携带 Cookie
}))

// 让 Express 能解析 JSON 格式的请求体
// 例如 POST /api/xxx 的 body: { "key": "value" }
app.use(express.json())

// ==================== 2. 健康检查接口 ====================

// 一个简单的 GET 接口，用来确认服务器是否在运行
// 访问 http://localhost:3000/api/health 就能看到 "ok"
app.get('/api/health', (_req, res) => {
  res.json({ status: 'ok', time: new Date().toISOString() })
})

// ==================== 3. 创建 HTTP + WebSocket 服务器 =========

const server = http.createServer(app)

// 让 WebSocket 监听同一个 HTTP 服务器
// 这样 WebSocket 和普通 HTTP 接口共用 3000 端口
createWsServer(server)

// ==================== 4. 启动服务器 ====================

// 端口号：优先用环境变量 PORT，没有就用 3000
const PORT = parseInt(process.env.PORT || '3000')

// 开始监听指定端口
// 回调函数在启动成功后执行
server.listen(PORT, () => {
  console.log(`🚀 HourMind 后端服务已启动`)
  console.log(`   HTTP:    http://localhost:${PORT}`)
  console.log(`   WebSocket: ws://localhost:${PORT}/ws`)
  console.log(`   健康检查: http://localhost:${PORT}/api/health`)
})
```

```typescript
// ============================================================
// wsServer.ts —— WebSocket 服务端
// WebSocket 和普通的 HTTP 不同：
//   HTTP:  一问一答，客户端发请求 → 服务器回应 → 连接关闭
//   WebSocket: 建立连接后可以随时互相发消息，连接保持
//
// 这里创建 WebSocket 服务，监听客户端连接和消息
// ============================================================

// 导入 ws 库中的 WebSocket 服务端类
import { WebSocketServer, WebSocket } from 'ws'
// 导入 Node.js 内置的 http 模块类型
import type { Server } from 'http'
// 导入 URL 解析工具（从查询参数中提取 token）
import { parse } from 'url'
// 导入 action 路由分发器
import { handleMessage } from './wsRouter'

export function createWsServer(httpServer: Server) {
  // ==================== 创建 WebSocket 服务 ====================
  // 把 WebSocket 服务"挂"在现有的 HTTP 服务器上
  // server 参数是客户端连接时的路径，这里设为 '/ws'
  // 客户端用 ws://localhost:3000/ws 连接
  const wss = new WebSocketServer({ server: httpServer, path: '/ws' })

  // ==================== 监听新连接 ====================
  // 每当有客户端连接时触发
  wss.on('connection', (ws: WebSocket, req) => {
    // --- 1. 从 URL 查询参数中提取 token ---
    // req.url 类似 "/ws?token=xxx"
    const urlParams = parse(req.url || '', true)
    const token = urlParams.query.token as string

    // MVP 阶段：如果还没有设置密码，允许无 token 连接
    // （后续在 auth 模块中会完善验证逻辑）
    console.log('🔌 新客户端连接成功', token ? `(已提供 Token)` : '(无 Token)')

    // --- 2. 监听客户端发来的消息 ---
    ws.on('message', async (rawData) => {
      try {
        // WebSocket 传的是 Buffer，先转成字符串
        const text = rawData.toString()
        // 再解析成 JSON 对象
        // 客户端发来的格式：{ id: "req_1", type: "request", action: "keys.list", payload: {} }
        const msg = JSON.parse(text)

        // --- 3. 只处理 type 为 "request" 的消息 ---
        if (msg.type === 'request') {
          // 调路由处理器执行业务逻辑
          // 传入 action 名称（如 "keys.list"）、payload 数据、token
          const response = await handleMessage(msg.action, msg.payload, token)

          // --- 4. 把处理结果返回给客户端 ---
          // 回复格式：{ id: "req_1", type: "response", success: true, data: {...} }
          ws.send(JSON.stringify({
            id: msg.id,          // 哪个请求的回复（客户端用这个 ID 匹配）
            type: 'response',
            success: response.success,
            data: response.data,
            error: response.error
          }))
        }
      } catch (err) {
        // 如果消息格式错误（比如不是合法 JSON），返回错误
        ws.send(JSON.stringify({
          id: '',
          type: 'response',
          success: false,
          error: { code: 'PARSE_ERROR', message: '消息格式错误' }
        }))
      }
    })

    // --- 5. 监听连接断开 ---
    ws.on('close', () => {
      console.log('🔌 客户端断开连接')
    })

    // --- 6. 发送一条欢迎消息 ---
    ws.send(JSON.stringify({
      id: '',
      type: 'response',
      success: true,
      data: { message: '连接成功', action: 'connected' }
    }))
  })

  console.log('✅ WebSocket 服务已就绪，路径: /ws')
}
```

```typescript
// ============================================================
// wsRouter.ts —— Action 路由分发器
// 这是后端的"交通指挥中心"
// 收到客户端消息后，根据 action 名称（如 "keys.list"）
// 转发给对应的处理函数去执行
// ============================================================

// 统一的响应格式
export interface WsResponse {
  success: boolean       // 操作是否成功
  data?: any             // 成功时的返回数据
  error?: {              // 失败时的错误信息
    code: string         //   错误码，如 "KEY_NOT_FOUND"
    message: string      //   人类可读的错误描述
  }
}

// 处理函数的类型定义
// 每个 handler 接收 payload（请求参数）和 token（用户认证），返回 WsResponse
type Handler = (payload: any, token?: string) => Promise<WsResponse>

// action 名称 → 处理函数的映射表
// 比如 "keys.list" 映射到 keyHandler.list
const routes: Record<string, Handler> = {}

// 注册路由：把 action 名称和处理函数关联起来
// 这个方法供后面的 handler 模块调用来注册自己
export function registerRoute(action: string, handler: Handler) {
  routes[action] = handler
}

// 根据 action 名称找到对应的 handler 并执行
export async function handleMessage(
  action: string,      // 客户端请求的 action，如 "keys.list"
  payload: any,        // 客户端传来的参数
  token?: string       // 客户端身份 token
): Promise<WsResponse> {
  // 查找是否注册了这个 action
  const handler = routes[action]

  // 如果没有注册，返回"未知操作"错误
  if (!handler) {
    return {
      success: false,
      error: {
        code: 'UNKNOWN_ACTION',
        message: `未知操作: ${action}`,
      },
    }
  }

  // 找到了就执行 handler，把结果返回
  return handler(payload, token)
}
```

```typescript
// ============================================================
// authHandler.ts —— 认证相关的处理函数
// 负责：
//   auth.setup  —— 首次设置密码
//   auth.login  —— 登录验证
//   auth.check  —— 确认 Token 是否有效
// ============================================================

// 导入数据库连接
import { prisma } from '../db'
// 导入 bcryptjs 用于密码哈希（明文密码不能存数据库）
import bcrypt from 'bcryptjs'
// 导入 jsonwebtoken 用于生成 JWT Token
import jwt from 'jsonwebtoken'
// 导入路由注册函数
import { registerRoute } from '../wsRouter'
import type { WsResponse } from '../wsRouter'

// JWT 密钥从环境变量读取
const JWT_SECRET = process.env.JWT_SECRET || 'fallback-secret'
// Token 有效期：24 小时
const TOKEN_EXPIRY = '24h'

// ==================== auth.setup ====================
// 首次使用：设置密码
// payload: { password: "用户输入的密码" }
registerRoute('auth.setup', async (payload): Promise<WsResponse> => {
  const { password } = payload

  // 1. 检查密码长度（至少 4 位，个人项目不要求太复杂）
  if (!password || password.length < 4) {
    return {
      success: false,
      error: { code: 'WEAK_PASSWORD', message: '密码至少需要 4 个字符' }
    }
  }

  // 2. 检查是否已经设置过密码
  // 查找 app_config 表中 key="password_hash" 的记录
  const existing = await prisma.appConfig.findUnique({
    where: { key: 'password_hash' }
  })

  if (existing) {
    return {
      success: false,
      error: { code: 'ALREADY_SETUP', message: '已经设置过密码了' }
    }
  }

  // 3. 把密码做哈希处理（Hash）
  // bcrypt.hash 把密码变成一串乱码，这个乱码推不出原始密码
  // 10 是"加密强度"，值越大越安全但也越慢
  const passwordHash = await bcrypt.hash(password, 10)

  // 4. 把哈希后的密码存到数据库
  // app_config 是 key-value 结构：key="password_hash", value=哈希值
  await prisma.appConfig.create({
    data: { key: 'password_hash', value: passwordHash }
  })

  // 5. 生成 JWT Token
  const token = jwt.sign(
    { setup: true },  // Token 里存储的信息（payload），这里简单存一个标记
    JWT_SECRET,       // 签名密钥
    { expiresIn: TOKEN_EXPIRY }  // 过期时间
  )

  // 6. 返回成功 + Token
  return {
    success: true,
    data: { token, message: '密码设置成功' }
  }
})

// ==================== auth.login ====================
// 登录：验证密码后返回 Token
// payload: { password: "用户输入的密码" }
registerRoute('auth.login', async (payload): Promise<WsResponse> => {
  const { password } = payload

  // 1. 从数据库取出存储的密码哈希
  const record = await prisma.appConfig.findUnique({
    where: { key: 'password_hash' }
  })

  // 2. 如果没有记录 = 还没设置密码
  if (!record) {
    return {
      success: false,
      error: { code: 'NOT_SETUP', message: '请先设置密码' }
    }
  }

  // 3. 比对用户输入的密码和数据库中的哈希
  // bcrypt.compare 内部会自动做哈希然后比对，不用手动哈希
  const isMatch = await bcrypt.compare(password, record.value)

  if (!isMatch) {
    return {
      success: false,
      error: { code: 'WRONG_PASSWORD', message: '密码错误' }
    }
  }

  // 4. 密码正确，生成新 Token
  const token = jwt.sign(
    { userId: 'default' },  // 单用户，userId 固定
    JWT_SECRET,
    { expiresIn: TOKEN_EXPIRY }
  )

  return {
    success: true,
    data: { token, message: '登录成功' }
  }
})

// ==================== auth.check ====================
// 验证 Token 是否有效
// 客户端每次打开页面时调用，确认之前的登录还没过期
registerRoute('auth.check', async (_payload, token): Promise<WsResponse> => {
  // 如果没有 token
  if (!token) {
    // 但已经设置过密码，说明需要重新登录
    const record = await prisma.appConfig.findUnique({
      where: { key: 'password_hash' }
    })
    if (record) {
      return {
        success: false,
        error: { code: 'TOKEN_REQUIRED', message: '请先登录' }
      }
    }
    // 没设置过密码，需要初始设置
    return {
      success: false,
      error: { code: 'NOT_SETUP', message: '请先设置密码' }
    }
  }

  try {
    // jwt.verify 验证 Token 是否有效
    // 如果 Token 被篡改或过期，会抛出异常
    jwt.verify(token, JWT_SECRET)
    return { success: true, data: { valid: true } }
  } catch {
    return {
      success: false,
      error: { code: 'INVALID_TOKEN', message: 'Token 无效或已过期' }
    }
  }
})
```

```typescript
// ============================================================
// cryptoService.ts —— AES 加密解密服务
// 把用户的 API Key 加密后存数据库，防止泄密
// 用的是 AES-256-GCM 算法（工业标准加密）
// ============================================================

// Node.js 内置的加密模块
import crypto from 'crypto'

// 加密密钥：从环境变量获取，换成 32 字节（AES-256 要求）
const KEY = Buffer.from(process.env.ENCRYPTION_KEY || 'hourmind-dev-key-32bytes-long!!', 'utf8')

// 加密函数：把明文 Key 变成密文
export function encrypt(plainText: string): { encrypted: string; suffix: string } {
  // 1. 生成一个随机的"初始向量"（IV），让每次加密结果都不一样
  //    即使加密同一个 Key，每次生成的密文也不同
  const iv = crypto.randomBytes(16)

  // 2. 创建加密器
  //    crypto.createCipheriv("算法名", "密钥", "初始向量")
  const cipher = crypto.createCipheriv('aes-256-gcm', KEY, iv)

  // 3. 加密数据
  let encrypted = cipher.update(plainText, 'utf8', 'hex')
  encrypted += cipher.final('hex')

  // 4. 获取"认证标签"（用于解密时验证数据是否被篡改）
  const authTag = cipher.getAuthTag()

  // 5. 把 iv + 认证标签 + 密文 拼接成一个字符串
  //    iv.toString('hex')  把 16 字节的 IV 转成 32 个十六进制字符
  //    authTag.toString('hex') 类似
  const combined = iv.toString('hex') + ':' + authTag.toString('hex') + ':' + encrypted

  // 6. 提取 Key 最后 6 位明文（用于列表里显示）
  const suffix = plainText.slice(-6)

  return { encrypted: combined, suffix }
}

// 解密函数：把数据库里的密文还原成明文 Key
export function decrypt(combined: string): string {
  // 1. 按照 ":" 分割数据
  const parts = combined.split(':')
  if (parts.length !== 3) {
    throw new Error('加密数据格式错误')
  }
  const [ivHex, authTagHex, encryptedHex] = parts

  // 2. 把十六进制字符串转回 Buffer
  const iv = Buffer.from(ivHex, 'hex')
  const authTag = Buffer.from(authTagHex, 'hex')

  // 3. 创建解密器
  const decipher = crypto.createDecipheriv('aes-256-gcm', KEY, iv)
  decipher.setAuthTag(authTag)

  // 4. 解密
  let decrypted = decipher.update(encryptedHex, 'hex', 'utf8')
  decrypted += decipher.final('utf8')

  return decrypted
}
```

```typescript
// ============================================================
// aiService.ts —— 调用 AI 厂商 API 的服务
// 由于 MVP 简化了技术栈，不做 Python 层
// Node.js 直接向 OpenAI/Anthropic 等厂商发起 HTTP 请求
// ============================================================

// 解密函数：从数据库取出加密的 Key 后需要解密才能使用
import { decrypt } from './cryptoService'

/**
 * 测试 Key 的连通性
 * 向厂商 API 发一个简单的请求，确认 Key 有效
 *
 * @param baseUrl - 厂商的 API 地址，如 "https://api.openai.com/v1"
 * @param encryptedKey - 数据库里存储的加密 Key
 * @returns { success: true/false, latencyMs: 响应时间 }
 */
export async function testKeyConnection(
  baseUrl: string,
  encryptedKey: string
): Promise<{ success: boolean; latencyMs: number; errorMessage?: string }> {
  // 1. 解密 Key（从数据库密文还原成真实 Key）
  let apiKey: string
  try {
    apiKey = decrypt(encryptedKey)
  } catch {
    return { success: false, latencyMs: 0, errorMessage: 'Key 解密失败' }
  }

  // 2. 记录开始时间（用来计算延迟）
  const startTime = Date.now()

  try {
    // 3. 向厂商 API 发请求
    //    用 models 端点测试连通性（轻量，不消耗 Token）
    const response = await fetch(`${baseUrl}/models`, {
      method: 'GET',
      headers: {
        'Authorization': `Bearer ${apiKey}`,   // 大多数厂商用 Bearer 认证
        'Content-Type': 'application/json',
      },
      // AbortSignal.timeout 是 Node.js 内置的：请求超时 10 秒自动取消
      signal: AbortSignal.timeout(10000),
    })

    // 4. 计算延迟（毫秒）
    const latencyMs = Date.now() - startTime

    // 5. 检查响应状态：200=成功，401=Key无效，403=权限不足
    if (response.ok) {
      return { success: true, latencyMs }
    } else {
      const body = await response.text()
      return {
        success: false,
        latencyMs,
        // 提取 API 返回的错误信息
        errorMessage: `HTTP ${response.status}: ${body.slice(0, 200)}`
      }
    }
  } catch (err: any) {
    // 6. 网络错误或超时
    const latencyMs = Date.now() - startTime
    return {
      success: false,
      latencyMs,
      errorMessage: err.name === 'TimeoutError'
        ? '连接超时（10 秒）'
        : `网络错误: ${err.message}`
    }
  }
}
```

```typescript
// ============================================================
// useWs.ts - WebSocket 客户端（全局单例，Promise 化）
// 用法：const keys = await wsClient.send('keys.list')
// ============================================================

// 等待回复的请求队列
const pending = new Map<string, {
  resolve: (d: any) => void; reject: (e: any) => void; timer: ReturnType<typeof setTimeout>
}>()
let counter = 0
let ws: WebSocket | null = null
const pushHandlers = new Map<string, Set<(d: any) => void>>()

// 连接 WebSocket
function connect(token: string): Promise<void> {
  return new Promise((resolve, reject) => {
    if (ws) { ws.close(); ws = null }
    ws = new WebSocket(`ws://localhost:3000/ws?token=${encodeURIComponent(token)}`)
    ws.onopen = () => resolve()
    ws.onerror = () => reject(new Error('连接失败'))
    ws.onmessage = (event) => {
      const msg = JSON.parse(event.data)
      if (msg.type === 'response') {
        const entry = pending.get(msg.id)
        if (entry) {
          clearTimeout(entry.timer)
          pending.delete(msg.id)
          msg.success ? entry.resolve(msg.data) : entry.reject(msg.error)
        }
        return
      }
      pushHandlers.get(msg.type)?.forEach(h => h(msg))
    }
    ws.onclose = () => {
      pending.forEach(e => { clearTimeout(e.timer); e.reject({ code: 'DISCONNECTED' }) })
      pending.clear()
    }
  })
}

// 发请求（返回 Promise）
function send<T>(action: string, payload: any = {}): Promise<T> {
  return new Promise((resolve, reject) => {
    const id = `req_${++counter}`
    const timer = setTimeout(() => { pending.delete(id); reject({ code: 'TIMEOUT' }) }, 10000)
    pending.set(id, { resolve, reject, timer })
    ws!.send(JSON.stringify({ id, type: 'request', action, payload }))
  })
}

// 注册推送监听
function onPush(type: string, h: (d: any) => void) {
  if (!pushHandlers.has(type)) pushHandlers.set(type, new Set())
  pushHandlers.get(type)!.add(h)
}
function offPush(type: string, h: (d: any) => void) {
  pushHandlers.get(type)?.delete(h)
}

export const wsClient = { connect, send, onPush, offPush }
```

```typescript
    isAuthenticated.value = true
  }

    const result = await wsClient.send('auth.login', { password })
    token.value = result.token
    localStorage.setItem('hourmind_token', result.token)
    isAuthenticated.value = true
  }

  function logout() {
    token.value = null
    isAuthenticated.value = false
    localStorage.removeItem('hourmind_token')
    window.location.reload()
  }

  async function checkAuth() {
    if (!token.value) {
      try {
        await wsClient.connect('')
        const result = await wsClient.send('auth.check', {})
        isAuthenticated.value = true
      } catch (err: any) {

      return
    }

      await wsClient.connect(token.value)
      const result = await wsClient.send('auth.check', {})
      if (result.valid) isAuthenticated.value = true
    } catch {
      token.value = null
      localStorage.removeItem('hourmind_token')
    }
  }
```

```typescript
  const loading = ref(false)
  const search = ref('')
  const statusFilter = ref('all')
  const stats = ref({ total: 0, active: 0, monthlyCost: 0 })
  const showAddDialog = ref(false)

  const filteredKeys = computed(() => {
    let result = keys.value
    if (statusFilter.value !== 'all') result = result.filter(k => k.status === statusFilter.value)
    if (search.value) {
      const kw = search.value.toLowerCase()
      result = result.filter(k => k.alias.toLowerCase().includes(kw) || k.provider.name.toLowerCase().includes(kw))
    }
    return result
  })

  async function fetchKeys() {
    loading.value = true
    try { keys.value = await wsClient.send('keys.list') }
    finally { loading.value = false }
  }

  async function fetchStats() { stats.value = await wsClient.send('keys.stats') }

    keys.value.unshift(result)
    await fetchStats()

    const result = await wsClient.send('keys.test', { keyId })
    const key = keys.value.find(k => k.id === keyId)

    return result
  }

    await wsClient.send('keys.delete', { keyId })
    keys.value = keys.value.filter(k => k.id !== keyId)
    await fetchStats()
  }

    await wsClient.send('keys.toggle', { keyId, enabled })
    const key = keys.value.find(k => k.id === keyId)
    if (key) key.status = enabled ? 'active' : 'disabled'
  }

  return { keys, loading, search, statusFilter, stats, showAddDialog, providers, filteredKeys, fetchKeys, fetchStats, fetchProviders, createKey, testKey, deleteKey, toggleKey }
```

- [ ] **Step 1：改造 wsRouter 支持推送**

```typescript
// ============================================================
// wsRouter.ts —— Action 路由分发器（改造版）
// handler 现在有两种模式：
//   1. 普通模式：return WsResponse（和之前一样）
//   2. 推送模式：通过 context.push() 多次推送消息
// ============================================================

import { WebSocket } from 'ws'

// 推送上下文：handler 用它来主动推消息给客户端
export class PushContext {
  constructor(private ws: WebSocket) {}

  // 推一条消息给客户端（不参与 request-response 匹配）
  push(type: string, data: any) {
    if (this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify({ type, ...data }))
    }
  }

  // 获取原始的 WebSocket 对象（供特殊场景使用）
  getRawWs(): WebSocket { return this.ws }
}

// 响应格式（不变）
export interface WsResponse {
  success: boolean
  data?: any
  error?: { code: string; message: string }
}

// Handler 签名升级：现在接收 PushContext 作为第三个参数
// 普通 handler 忽略 ctx 直接 return
// 推送 handler 用 ctx.push() 推消息，然后 return { success: true }
type Handler = (payload: any, token?: string, ctx?: PushContext) => Promise<WsResponse>

const routes: Record<string, Handler> = {}

export function registerRoute(action: string, handler: Handler) {
  routes[action] = handler
}

export async function handleMessage(
  action: string, payload: any, token: string | undefined, ws: WebSocket
): Promise<WsResponse> {
  const handler = routes[action]
  if (!handler) {
    return { success: false, error: { code: 'UNKNOWN_ACTION', message: `未知操作: ${action}` } }
  }
  // 创建推送上下文传给 handler
  const ctx = new PushContext(ws)
  return handler(payload, token, ctx)
}
```

- [ ] **Step 1：添加流式对话函数到 aiService**

在 `hourmind/server/src/services/aiService.ts` 文件末尾追加：

```typescript
// ============================================================
// 流式对话（Streaming Chat）
// 调 AI 厂商的 chat/completions 接口，获取逐 token 的流式响应
// 
// SSE（Server-Sent Events）格式说明：
//   厂商返回的数据是以 "data: " 开头的行
//   每行一个 JSON 对象，包含增量文本
//   "data: [DONE]" 表示流结束
// ============================================================

/**
 * 流式聊天请求
 *
 * @param baseUrl    - 厂商 API 地址
 * @param encryptedKey - 加密的 API Key（会内部解密）
 * @param model      - 模型名，如 "gpt-4o", "deepseek-chat"
 * @param messages   - 消息数组 [{role:"user", content:"你好"}]
 * @param onChunk    - 每收到一个 token 就调一次：onChunk("你好")
 * @param onDone     - 流结束回调：onDone(总Token数)
 * @param onError    - 出错回调：onError("错误描述")
 * @returns 返回一个 abort 函数，调用它可以取消请求
 */
export function streamChat(
  baseUrl: string,
  encryptedKey: string,
  model: string,
  messages: { role: string; content: string }[],
  onChunk: (text: string) => void,
  onDone: (totalTokens: number) => void,
  onError: (error: string) => void,
): () => void {
  // 1. 解密 Key
  let apiKey: string
  try {
    apiKey = decrypt(encryptedKey)
  } catch {
    onError('Key 解密失败')
    return () => {}
  }

  // 2. 创建 AbortController（用于取消请求）
  const controller = new AbortController()

  // 3. 异步执行流式请求（不阻塞主线程）
  ;(async () => {
    try {
      // 发起 HTTP POST 请求
      const response = await fetch(`${baseUrl}/chat/completions`, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${apiKey}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          model,
          messages,
          stream: true,  // ⭐ 关键参数：启用流式输出
        }),
        signal: controller.signal,  // 关联 AbortController
      })

      // 如果 HTTP 状态码不是 2xx，说明请求失败
      if (!response.ok) {
        const body = await response.text()
        onError(`API 错误 (${response.status}): ${body.slice(0, 300)}`)
        return
      }

      // 4. 获取流读取器
      //    response.body 是一个 ReadableStream（字节流）
      const reader = response.body!.getReader()
      const decoder = new TextDecoder()  // 把字节解码成 UTF-8 字符串
      let totalTokens = 0
      let buffer = ''  // 缓冲区：处理被截断的 SSE 行

      // 5. 循环读取数据块
      while (true) {
        const { done, value } = await reader.read()
        if (done) break  // done=true 表示流结束了

        // 把新收到的字节追加到缓冲区
        buffer += decoder.decode(value, { stream: true })

        // 按行分割（SSE 协议中每条消息占一行）
        const lines = buffer.split('\n')
        // 最后一行可能不完整，留在缓冲区里下次再处理
        buffer = lines.pop() || ''

        for (const line of lines) {
          const trimmed = line.trim()
          if (!trimmed || !trimmed.startsWith('data: ')) continue

          const dataStr = trimmed.slice(6)  // 去掉 "data: " 前缀
          if (dataStr === '[DONE]') continue  // 流结束标记

          try {
            const data = JSON.parse(dataStr)
            const delta = data.choices?.[0]?.delta

            // 如果 delta 里有 content，就推送给调用方
            if (delta?.content) {
              onChunk(delta.content)
            }

            // 记录最后的 Token 统计
            if (data.usage?.total_tokens) {
              totalTokens = data.usage.total_tokens
            }
          } catch {
            // JSON 解析失败就跳过（某些厂商格式不标准）
          }
        }
      }

      // 6. 流正常结束
      onDone(totalTokens)
    } catch (err: any) {
      // 如果是主动取消，不报错
      if (err.name === 'AbortError') return
      if (err.name === 'TimeoutError') {
        onError('请求超时（2分钟）')
      } else {
        onError(`网络错误: ${err.message}`)
      }
    }
  })()

  // 返回取消函数
  return () => controller.abort()
}
```

```typescript
  return { success: true, data: { message: '已更新' } }

// ==================== messages.list ====================
// 获取某个会话的所有消息（按时间正序）
// payload: { conversationId }
registerRoute('messages.list', async (payload): Promise<WsResponse> => {
  const { conversationId } = payload
  const conv = await prisma.conversation.findUnique({
    where: { id: conversationId },
    include: { messages: { orderBy: { createdAt: 'asc' } } }
  })
  if (!conv) {
    return { success: false, error: { code: 'CONVERSATION_NOT_FOUND', message: '会话不存在' } }
  }
  return { success: true, data: { conversation: conv, messages: conv.messages } }
})

// ==================== messages.send（核心中的核心）============
// 发送消息并以流式方式获取 AI 回复
//
// 前端收到初始回复后，需要注册 push 监听：
//   wsClient.onPush('stream_chunk', handler)
//   wsClient.onPush('stream_end', handler)
//   wsClient.onPush('stream_error', handler)
//
// payload: { conversationId, content, model? }
registerRoute('messages.send', async (payload, _token, ctx): Promise<WsResponse> => {
  const { conversationId, content, model } = payload

  // --- 1. 查找会话 ---
  const conversation = await prisma.conversation.findUnique({
    where: { id: conversationId }
  })
  if (!conversation) {
    return { success: false, error: { code: 'CONVERSATION_NOT_FOUND', message: '会话不存在' } }
  }

  // --- 2. 找到可用的 API Key ---
  const apiKey = await prisma.apiKey.findFirst({
    where: { status: 'active', isDeleted: false },
    include: { provider: true },
  })
  if (!apiKey) {
    return { success: false, error: { code: 'NO_ACTIVE_KEY', message: '没有可用的 API Key，请先添加' } }
  }

  // --- 3. 保存用户消息到数据库 ---
  const userMsg = await prisma.message.create({
    data: { conversationId, role: 'user', content }
  })

  // --- 4. 创建空的 assistant 消息（流式结束后再更新内容）---
  const currentModel = model || conversation.model || 'gpt-4o'
  const assistantMsg = await prisma.message.create({
    data: { conversationId, role: 'assistant', content: '', model: currentModel }
  })

  // 更新会话的信息
  await prisma.conversation.update({
    where: { id: conversationId },
    data: {
      messageCount: { increment: 2 },    // 用户 + AI 各一条
      updatedAt: new Date(),
      // 如果是第一条消息，用用户消息前 30 字做标题
      ...(conversation.messageCount === 0 ? { title: content.slice(0, 30) } : {}),
    }
  })

  // --- 5. 先回复客户端，告诉它流式即将开始 ---
  //    这个回复走普通的 response 通道
  //    return 之后 handler 就返回了，但流还在后台继续

  // --- 6. 获取历史消息构建上下文 ---
  const history = await prisma.message.findMany({
    where: { conversationId },
    orderBy: { createdAt: 'asc' },
    take: 40,  // 取最近 40 条作为上下文
  })

  const apiMessages = history.map(m => ({
    role: m.role === 'assistant' ? 'assistant' : 'user',
    content: m.content,
  }))

  // --- 7. 异步启动流式调用（不阻塞 handler 返回）---
  const fullContent: string[] = []  // 用数组收集所有 chunk

  streamChat(
    apiKey.provider.baseUrl,
    apiKey.encryptedKey,
    currentModel,
    apiMessages,
    // onChunk：每收到一小段文本就推给前端
    (chunk: string) => {
      fullContent.push(chunk)
      ctx?.push('stream_chunk', {
        message_id: assistantMsg.id,
        chunk,
      })
    },
    // onDone：流结束后更新数据库
    async (totalTokens: number) => {
      // 更新消息内容和 Token 数
      const content = fullContent.join('')
      await prisma.message.update({
        where: { id: assistantMsg.id },
        data: { content, tokenCount: totalTokens }
      })

      // 更新会话累计 Token
      await prisma.conversation.update({
        where: { id: conversationId },
        data: { totalTokens: { increment: totalTokens } }
      })

      // 记录用量
      await prisma.tokenUsageRecord.create({
        data: {
          keyId: apiKey.id,
          modelName: currentModel,
          promptTokens: 0,   // 流式响应通常给总 token，不区分 prompt/completion
          completionTokens: 0,
          estimatedCostCents: 0,
          sessionId: conversationId,
        }
      })

      // 推送流结束帧
      ctx?.push('stream_end', {
        message_id: assistantMsg.id,
        token_count: totalTokens,
        model: currentModel,
      })
    },
    // onError：出错时推送错误帧
    (error: string) => {
      // 更新消息内容为错误信息
      prisma.message.update({
        where: { id: assistantMsg.id },
        data: { content: `[错误] ${error}` }
      }).catch(() => {})

      ctx?.push('stream_error', {
        message_id: assistantMsg.id,
        error: { code: 'STREAM_ERROR', message: error },
      })
    },
  )

  // --- 8. 立即返回，告诉前端流式要开始了 ---
  return {
    success: true,
    data: {
      mode: 'stream',
      user_message_id: userMsg.id,
      assistant_message_id: assistantMsg.id,
      model: currentModel,
    }
  }
})

// ==================== messages.stop ====================
// 停止生成（用户按了 Esc 或停止按钮）

// 暂时不做取消功能，等后续完善

  // TODO: 后续通过存储 abort controllers 来实现真正的取消
```

```typescript
// ============================================================
// chatStore.ts —— 对话状态管理
// 管理：左侧会话列表 + 中间消息列表 + 流式输出状态
// ============================================================

import { defineStore } from 'pinia'
import { ref, nextTick } from 'vue'
import { wsClient } from '@/composables/useWs'

// 类型定义
export interface Conversation {
  id: string; title: string; model: string; updatedAt: string; messageCount: number
}
export interface Message {
  id: string; role: 'user' | 'assistant'; content: string; model?: string; tokenCount?: number
}

export const useChatStore = defineStore('chat', () => {
  // === 状态 ===
  const conversations = ref<Conversation[]>([])
  const activeId = ref<string | null>(null)          // 当前打开的是哪个会话
  const messages = ref<Message[]>([])
  const loading = ref(false)
  const isStreaming = ref(false)                     // 是否正在接收 AI 回复
  const currentModel = ref('gpt-4o')                 // 当前使用的模型

  // === 会话操作 ===

  // 拉取会话列表
  async function fetchConversations() {
    loading.value = true
    try { conversations.value = await wsClient.send('conversations.list') }
    finally { loading.value = false }
  }

  // 新建会话
  async function createConversation(): Promise<string> {
    const result = await wsClient.send('conversations.create', {
      title: '新对话', model: currentModel.value
    })
    conversations.value.unshift(result)
    activeId.value = result.id
    messages.value = []
    return result.id
  }

  // 选中一个会话，加载它的消息
  async function selectConversation(id: string) {
    activeId.value = id
    loading.value = true
    try {
      const result = await wsClient.send('messages.list', { conversationId: id })
      messages.value = result.messages || []
      // 滚动到底部（等 Vue 渲染完再滚）
      await nextTick()
      scrollToBottom()
    } finally {
      loading.value = false
    }
  }

  // 删除会话
  async function deleteConversation(id: string) {
    await wsClient.send('conversations.delete', { conversationId: id })
    conversations.value = conversations.value.filter(c => c.id !== id)
    if (activeId.value === id) {
      activeId.value = null
      messages.value = []
    }
  }

  // === 消息操作 ===

  // 发送消息（核心中的核心）
  async function sendMessage(content: string) {
    // 如果没有活跃会话，先创建
    if (!activeId.value) await createConversation()
    const convId = activeId.value!

    // 1. 乐观更新：立刻在界面上显示用户消息
    messages.value.push({
      id: 'u_' + Date.now(), role: 'user', content
    })

    // 2. 插入一条空的 AI 消息（占位，等待流式填充）
    const aiId = 'a_' + Date.now()
    messages.value.push({
      id: aiId, role: 'assistant', content: ''
    })

    // 滚动到底部
    await nextTick(); scrollToBottom()

    // 3. 发请求启动流式
    isStreaming.value = true

    try {
      const init = await wsClient.send('messages.send', {
        conversationId: convId,
        content,
        model: currentModel.value,
      })

      // 如果返回的是流式模式
      if (init.mode === 'stream') {
        // 把临时 ID 替换成服务器分配的真实 ID
        const aiMsg = messages.value.find(m => m.id === aiId)
        if (aiMsg) {
          aiMsg.id = init.assistant_message_id
          aiMsg.model = init.model
        }

        // 监听 stream_chunk（每收到一小段文本就追加到内容后）
        const chunkH = (msg: any) => {
          if (msg.message_id === init.assistant_message_id) {
            const m = messages.value.find(x => x.id === init.assistant_message_id)
            if (m) m.content += msg.chunk
            scrollToBottom()
          }
        }

        // 监听 stream_end（流结束）
        const endH = (msg: any) => {
          if (msg.message_id === init.assistant_message_id) {
            isStreaming.value = false
            const m = messages.value.find(x => x.id === init.assistant_message_id)
            if (m && msg.token_count) m.tokenCount = msg.token_count
            // 清理监听器
            wsClient.offPush('stream_chunk', chunkH)
            wsClient.offPush('stream_end', endH)
            wsClient.offPush('stream_error', errH)
            // 刷新会话列表（更新标题和时间）
            fetchConversations()
          }
        }

        // 监听 stream_error（流出错）
        const errH = (msg: any) => {
          if (msg.message_id === init.assistant_message_id) {
            isStreaming.value = false
            const m = messages.value.find(x => x.id === init.assistant_message_id)
            if (m) m.content = `[错误] ${msg.error?.message || '未知错误'}`
            wsClient.offPush('stream_chunk', chunkH)
            wsClient.offPush('stream_end', endH)
            wsClient.offPush('stream_error', errH)
          }
        }

        // 注册监听
        wsClient.onPush('stream_chunk', chunkH)
        wsClient.onPush('stream_end', endH)
        wsClient.onPush('stream_error', errH)
      }
    } catch (err: any) {
      isStreaming.value = false
      const m = messages.value.find(x => x.id === aiId)
      if (m) m.content = `[错误] ${err.message || '发送失败'}`
    }
  }

  // ==================== 辅助 ====================

  // 滚动消息列表到底部
  function scrollToBottom() {
    // 等一帧（16ms）让 DOM 更新完
    setTimeout(() => {
      const el = document.getElementById('chat-messages')
      if (el) el.scrollTop = el.scrollHeight
    }, 16)
  }

  return {
    conversations, activeId, messages, loading, isStreaming, currentModel,
    fetchConversations, createConversation, selectConversation, deleteConversation,
    sendMessage,
  }
})
```

```vue
<script setup lang="ts">
defineProps<{
  message: { id: string; role: string; content: string; tokenCount?: number }
}>()

// 简单的文本格式化：换行 → <br>，处理一下基本格式
function formatContent(text: string): string {
  if (!text) return ''
  // 把 HTML 特殊字符转义（防止 XSS 攻击）
  let safe = text
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
  // 把 \n 换行转成 <br>
  safe = safe.replace(/\n/g, '<br>')
  return safe
}
```

创建 `hourmind/client/src/components/ChatInput.vue`：

```vue
<!-- ============================================================
 ChatInput.vue —— 底部输入栏
   大输入框 + 发送按钮，支持 Enter 发送，Shift+Enter 换行
============================================================ -->
<template>
  <div class="chat-input-area">
    <div class="input-row">
      <!-- 多行文本输入框 -->
      <textarea
        ref="inputRef"
        v-model="text"
        class="msg-input"
        placeholder="输入消息... (Enter 发送，Shift+Enter 换行)"
        rows="1"
        :disabled="disabled"
        @keydown="onKeydown"
        @input="autoResize"
      ></textarea>

      <!-- 发送按钮 -->
      <button
        class="btn-primary send-btn"
        :disabled="disabled || !text.trim()"
        @click="send"
      >发送</button>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, nextTick, watch } from 'vue'

const props = defineProps<{ disabled: boolean }>()
const emit = defineEmits<{ send: [content: string] }>()

const text = ref('')
const inputRef = ref<HTMLTextAreaElement | null>(null)

// 按键处理：Enter 发送（不带 Shift），Shift+Enter 换行
function onKeydown(e: KeyboardEvent) {
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault()    // 阻止默认的换行行为
    send()
  }
}

// 发送消息
function send() {
  const content = text.value.trim()
  if (!content) return
  emit('send', content)
  text.value = ''
  // 重置输入框高度
  nextTick(() => {
    if (inputRef.value) {
      inputRef.value.style.height = 'auto'
    }
  })
}

// 自动调整输入框高度（内容多了自动变高）
function autoResize() {
  const el = inputRef.value
  if (!el) return
  el.style.height = 'auto'
  el.style.height = Math.min(el.scrollHeight, 200) + 'px'  // 最高 200px
}

// 流式结束时自动聚焦输入框
watch(() => props.disabled, (val) => {
  if (!val) inputRef.value?.focus()
})
</script>
```
