# server/services/ai_service.py
# AI 厂商 API 调用服务 —— 流式聊天补全，解析 SSE 逐块 yield
import json
import httpx
from database import get_db
from services.crypto_service import decrypt


async def stream_chat(conversation_id: str, model: str, content: str, cancel_flag: dict = None):
    """
    流式调用 AI 厂商 API，逐块返回文本
    - cancel_flag: 传入一个 dict，设置 cancel_flag["cancel"] = True 可中断流式输出
    - 逐块 yield dict: {"type": "chunk", "content": "..."}
    - 结束时 yield: {"type": "end", "token_count": N, "model": "..."}
    - 出错时 yield: {"type": "error", "message": "..."}
    """
    db = get_db()

    try:
        # 1. 查找支持该模型的活跃 API Key（按 provider 的 is_active + key 的 status 筛选）
        key_row = db.execute(
            """SELECT ak.encrypted_key, ak.id AS key_id, p.base_url, p.slug
               FROM api_key ak
               JOIN provider p ON ak.provider_id = p.id
               WHERE ak.status = 'active' AND p.is_active = 1
               ORDER BY ak.updated_at DESC LIMIT 1"""
        ).fetchone()

        if not key_row:
            yield {"type": "error", "message": "没有可用的 API Key，请先在 Keys 页面添加"}
            return

        # 2. 解密 API Key
        decrypted_key = decrypt(key_row["encrypted_key"])

        # 3. 获取 base_url，确保末尾没有斜杠
        base_url = key_row["base_url"].rstrip("/")

        # 4. 构建 messages 数组 —— 从 conversation 表中读取历史消息
        conv_row = db.execute(
            "SELECT messages_json FROM conversation WHERE id = ?",
            (conversation_id,)
        ).fetchone()

        history_msgs = []
        if conv_row:
            history_msgs = json.loads(conv_row["messages_json"])

        # 将历史消息转为 API 格式（只保留 role 和 content）
        messages = [{"role": m["role"], "content": m["content"]} for m in history_msgs]

        # 5. 发起流式 POST 请求
        headers = {
            "Authorization": f"Bearer {decrypted_key}",
            "Content-Type": "application/json",
        }
        body = {
            "model": model,
            "messages": messages,
            "stream": True,
        }

        full_content = ""
        token_count = 0

        async with httpx.AsyncClient(timeout=httpx.Timeout(120.0)) as client:
            async with client.stream(
                "POST",
                f"{base_url}/chat/completions",
                headers=headers,
                json=body,
            ) as response:
                if response.status_code != 200:
                    error_text = await response.aread()
                    yield {"type": "error", "message": f"API 错误 ({response.status_code}): {error_text.decode()[:200]}"}
                    return

                # 6. 逐行解析 SSE
                async for line in response.aiter_lines():
                    # 检查取消标志
                    if cancel_flag and cancel_flag.get("cancel"):
                        yield {"type": "error", "message": "用户取消了生成"}
                        return

                    if not line or not line.startswith("data: "):
                        continue

                    data_str = line[6:]  # 去掉 "data: " 前缀

                    if data_str.strip() == "[DONE]":
                        break

                    try:
                        chunk = json.loads(data_str)
                        choices = chunk.get("choices", [])
                        if choices:
                            delta = choices[0].get("delta", {})
                            content_chunk = delta.get("content", "")
                            if content_chunk:
                                full_content += content_chunk
                                yield {"type": "chunk", "content": content_chunk}
                    except json.JSONDecodeError:
                        continue

                # 7. 估算 token 数量（粗略：字符数 / 2，后续可用 tiktoken 精确计算）
                token_count = max(1, len(full_content) // 2)

                # 8. 更新 conversation 表：追加 assistant 消息并更新统计
                now = __import__("datetime").datetime.utcnow().isoformat() + "Z"
                assistant_msg = {
                    "id": f"msg_{token_count}_{int(__import__('time').time())}",
                    "role": "assistant",
                    "content": full_content,
                    "model": model,
                    "token_count": token_count,
                    "created_at": now,
                }

                db.execute(
                    """UPDATE conversation
                       SET messages_json = json_insert(messages_json, '$[#]', ?),
                           total_tokens = total_tokens + ?,
                           message_count = message_count + 1,
                           updated_at = ?
                       WHERE id = ?""",
                    (json.dumps(assistant_msg), token_count, now, conversation_id),
                )
                db.commit()

                yield {"type": "end", "token_count": token_count, "model": model}

    except httpx.ConnectError:
        yield {"type": "error", "message": "无法连接到 AI 厂商服务器"}
    except httpx.TimeoutException:
        yield {"type": "error", "message": "AI 厂商响应超时"}
    except Exception as e:
        yield {"type": "error", "message": f"流式调用异常: {str(e)}"}
    finally:
        db.close()
```

# server/routes/conversations.py
# 会话 REST API —— 列表、创建、删除、编辑、详情
import json
import uuid
from datetime import datetime
from fastapi import APIRouter, Depends, Query, HTTPException
from database import get_db
from auth import require_auth

router = APIRouter(prefix="/api/conversations", tags=["conversations"])


@router.get("")
def list_conversations(
    status: str = Query("active", description="筛选状态：active | archived | deleted"),
    page: int = Query(1, ge=1, description="页码，从 1 开始"),
    token: str = Depends(require_auth),
):
    """获取会话列表，按 updated_at 倒序，支持分页
    - 默认 page_size = 20
    - 返回 { items[], total, page, page_size }
    """
    db = get_db()
    page_size = 20
    offset = (page - 1) * page_size

    try:
        # 查询总数
        total_row = db.execute(
            "SELECT COUNT(*) AS cnt FROM conversation WHERE status = ?",
            (status,),
        ).fetchone()
        total = total_row["cnt"] if total_row else 0

        # 查询分页数据
        rows = db.execute(
            """SELECT id, title, model, tags_json, is_pinned, is_starred,
                      status, summary, total_tokens, message_count,
                      created_at, updated_at
               FROM conversation
               WHERE status = ?
               ORDER BY is_pinned DESC, updated_at DESC
               LIMIT ? OFFSET ?""",
            (status, page_size, offset),
        ).fetchall()

        items = []
        for row in rows:
            items.append({
                "id": row["id"],
                "title": row["title"],
                "model": row["model"],
                "tags": json.loads(row["tags_json"] or "[]"),
                "is_pinned": bool(row["is_pinned"]),
                "is_starred": bool(row["is_starred"]),
                "status": row["status"],
                "summary": row["summary"],
                "total_tokens": row["total_tokens"],
                "message_count": row["message_count"],
                "created_at": row["created_at"],
                "updated_at": row["updated_at"],
            })

        return {
            "items": items,
            "total": total,
            "page": page,
            "page_size": page_size,
        }
    finally:
        db.close()


@router.post("")
def create_conversation(
    payload: dict,
    token: str = Depends(require_auth),
):
    """创建新会话
    - 请求体: { title, model }
    - 返回创建的会话对象
    """
    db = get_db()
    conv_id = str(uuid.uuid4())
    now = datetime.utcnow().isoformat() + "Z"
    title = payload.get("title", "新对话")
    model = payload.get("model", "deepseek")

    try:
        db.execute(
            """INSERT INTO conversation (id, title, model, created_at, updated_at)
               VALUES (?, ?, ?, ?, ?)""",
            (conv_id, title, model, now, now),
        )
        db.commit()

        return {
            "id": conv_id,
            "title": title,
            "model": model,
            "tags": [],
            "is_pinned": False,
            "is_starred": False,
            "status": "active",
            "summary": None,
            "total_tokens": 0,
            "message_count": 0,
            "created_at": now,
            "updated_at": now,
        }
    finally:
        db.close()


@router.delete("/{conv_id}")
def delete_conversation(
    conv_id: str,
    token: str = Depends(require_auth),
):
    """软删除会话 —— 将 status 设为 'deleted'"""
    db = get_db()
    try:
        db.execute(
            "UPDATE conversation SET status = 'deleted', updated_at = ? WHERE id = ?",
            (datetime.utcnow().isoformat() + "Z", conv_id),
        )
        db.commit()
        return {"ok": True}
    finally:
        db.close()


@router.put("/{conv_id}")
def update_conversation(
    conv_id: str,
    payload: dict,
    token: str = Depends(require_auth),
):
    """编辑会话 —— 可更新 title、is_pinned、is_starred、status
    - 请求体: { title?, is_pinned?, is_starred?, status? }
    """
    db = get_db()
    now = datetime.utcnow().isoformat() + "Z"
    allowed = {"title", "is_pinned", "is_starred", "status"}
    updates = {k: v for k, v in payload.items() if k in allowed}

    if not updates:
        raise HTTPException(status_code=400, detail="没有可更新的字段")

    try:
        # 构建动态 UPDATE
        set_clauses = ", ".join(f"{k} = ?" for k in updates)
        values = list(updates.values()) + [now, conv_id]
        db.execute(
            f"UPDATE conversation SET {set_clauses}, updated_at = ? WHERE id = ?",
            values,
        )
        db.commit()
        return {"ok": True}
    finally:
        db.close()


@router.get("/{conv_id}")

            "tags": json.loads(row["tags_json"] or "[]"),
            "is_pinned": bool(row["is_pinned"]),
            "is_starred": bool(row["is_starred"]),
            "status": row["status"],
            "summary": row["summary"],
            "total_tokens": row["total_tokens"],
            "message_count": row["message_count"],
            "created_at": row["created_at"],
            "updated_at": row["updated_at"],

        db.close()

# server/ws/chat.py
# 聊天 WebSocket 端点 —— 接收用户消息，调用 AI 流式服务，逐块推回前端
import json
import uuid
import asyncio
from datetime import datetime
from fastapi import WebSocket, WebSocketDisconnect, Query
from database import get_db
from auth import verify_token  # JWT 验证函数
from services.ai_service import stream_chat


async def chat_ws(websocket: WebSocket, token: str = Query(...)):
    """WebSocket 聊天端点 /ws/chat
    - token 参数: JWT 认证令牌（通过 query string 传递）
    - 前端发送 JSON: { action, conversation_id?, content?, model? }
    - 服务端推送 JSON: { type, message_id?, content?, ... }
    """
    # 验证 JWT
    payload = verify_token(token)
    if payload is None:
        await websocket.close(code=4001, reason="认证失败")
        return

    await websocket.accept()

    # 当前流式任务和取消标志
    current_task: asyncio.Task | None = None
    cancel_flag = {"cancel": False}

    try:
        while True:
            data = await websocket.receive_json()
            action = data.get("action")

            if action == "send":
                conv_id = data.get("conversation_id")
                content = data.get("content", "").strip()
                model = data.get("model", "deepseek")

                if not conv_id or not content:
                    await websocket.send_json({"type": "error", "message": "缺少 conversation_id 或 content"})
                    continue

                db = get_db()
                now = datetime.utcnow().isoformat() + "Z"

                try:
                    # 1. 保存用户消息到 conversation 的 messages_json
                    user_msg_id = f"msg_{uuid.uuid4().hex[:12]}"
                    user_msg = {
                        "id": user_msg_id,
                        "role": "user",
                        "content": content,
                        "created_at": now,
                    }
                    db.execute(
                        """UPDATE conversation
                           SET messages_json = json_insert(messages_json, '$[#]', ?),
                               updated_at = ?
                           WHERE id = ?""",
                        (json.dumps(user_msg), now, conv_id),
                    )
                    db.commit()
                finally:
                    db.close()

                # 2. 通知前端开始生成
                assistant_msg_id = f"msg_{uuid.uuid4().hex[:12]}"
                await websocket.send_json({
                    "type": "start",
                    "message_id": assistant_msg_id,
                })

                # 3. 重置取消标志，启动流式调用
                cancel_flag["cancel"] = False

                async def stream_to_client():
                    """异步任务：从 AI 服务读取 chunk 并推送到 WebSocket"""
                    async for chunk in stream_chat(conv_id, model, content, cancel_flag):
                        chunk["message_id"] = assistant_msg_id
                        await websocket.send_json(chunk)

                current_task = asyncio.create_task(stream_to_client())
                try:
                    await current_task
                except asyncio.CancelledError:
                    pass  # 被 stop 操作取消

            elif action == "stop":
                # 取消当前流式任务
                cancel_flag["cancel"] = True
                if current_task and not current_task.done():
                    current_task.cancel()
                await websocket.send_json({
                    "type": "stopped",
                    "message": "生成已停止",
                })

            elif action == "regenerate":
                conv_id = data.get("conversation_id")
                if not conv_id:
                    await websocket.send_json({"type": "error", "message": "缺少 conversation_id"})
                    continue

                db = get_db()
                try:
                    # 删除最后一条 assistant 消息
                    conv = db.execute(
                        "SELECT messages_json FROM conversation WHERE id = ?",
                        (conv_id,),
                    ).fetchone()

                    if conv:
                        messages = json.loads(conv["messages_json"])
                        # 找到最后一条 assistant 消息并移除
                        for i in range(len(messages) - 1, -1, -1):
                            if messages[i]["role"] == "assistant":
                                messages.pop(i)
                                break

                        # 提取最后一条 user 消息内容用于重新生成
                        last_user_content = ""
                        for i in range(len(messages) - 1, -1, -1):
                            if messages[i]["role"] == "user":
                                last_user_content = messages[i]["content"]
                                break

                        db.execute(
                            "UPDATE conversation SET messages_json = ?, updated_at = ? WHERE id = ?",
                            (json.dumps(messages), datetime.utcnow().isoformat() + "Z", conv_id),
                        )
                        db.commit()
                finally:
                    db.close()

                if last_user_content:
                    # 重新生成 —— 复用 send 逻辑（先不保存用户消息，因为已经在历史中）
                    assistant_msg_id = f"msg_{uuid.uuid4().hex[:12]}"
                    await websocket.send_json({
                        "type": "start",
                        "message_id": assistant_msg_id,
                    })

                    cancel_flag["cancel"] = False

                    async def regenerate_stream():
                        async for chunk in stream_chat(conv_id, model, last_user_content, cancel_flag):
                            chunk["message_id"] = assistant_msg_id
                            await websocket.send_json(chunk)

                    current_task = asyncio.create_task(regenerate_stream())
                    try:
                        await current_task
                    except asyncio.CancelledError:
                        pass

    except WebSocketDisconnect:
        # 客户端断开，取消正在进行的流式任务
        cancel_flag["cancel"] = True
        if current_task and not current_task.done():
            current_task.cancel()
    except Exception as e:
        # 未知错误，尝试通知客户端
        try:
            await websocket.send_json({"type": "error", "message": f"服务端异常: {str(e)}"})
        except Exception:
            pass
```

# server/ws/system.py
# 系统 WebSocket 端点 —— 认证 + 保持连接，留作后续推送通知用
from fastapi import WebSocket, WebSocketDisconnect, Query
from auth import verify_token


async def system_ws(websocket: WebSocket, token: str = Query(...)):
    """系统通知 WebSocket 端点 /ws
    - 验证 JWT，保活连接
    - 当前为占位实现，后续扩展通知推送（任务提醒、Token 告警等）
    """
    # 验证 JWT
    payload = verify_token(token)
    if payload is None:
        await websocket.close(code=4001, reason="认证失败")
        return

    await websocket.accept()

    # 发送连接确认
    await websocket.send_json({"type": "connected", "message": "系统通知通道已连接"})

    try:
        while True:
            # 等待客户端消息（ping/pong 保持连接）
            data = await websocket.receive_text()
            # 收到任何消息回复 pong
            if data == "ping":
                await websocket.send_text("pong")
    except WebSocketDisconnect:
        pass

// client/src/composables/useWs.ts
// WebSocket 客户端 —— 单例连接，支持消息收发和自动重连
import { ref, onUnmounted } from 'vue'

// 全局单例状态
let ws: WebSocket | null = null
const listeners = new Set<(data: any) => void>()
const isConnected = ref(false)
let reconnectTimer: ReturnType<typeof setTimeout> | null = null
let reconnectAttempts = 0
const MAX_RECONNECT_ATTEMPTS = 10
const RECONNECT_DELAY_MS = 2000

/**
 * 获取当前连接状态
 */
export function useWsStatus() {
  return { isConnected }
}

/**
 * 初始化 WebSocket 连接
 * @param token - JWT 认证令牌
 */
export function connectWs(token: string): void {
  // 如果已有连接且状态正常，先关闭
  if (ws && (ws.readyState === WebSocket.OPEN || ws.readyState === WebSocket.CONNECTING)) {
    return
  }

  // 构建 WebSocket URL（Vite 开发服务器代理 /ws 到后端）
  const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:'
  const wsUrl = `${protocol}//${window.location.host}/ws/chat?token=${encodeURIComponent(token)}`

  ws = new WebSocket(wsUrl)

  ws.onopen = () => {
    isConnected.value = true
    reconnectAttempts = 0
    console.log('[WS] 聊天 WebSocket 已连接')
  }

  ws.onmessage = (event: MessageEvent) => {
    try {
      const data = JSON.parse(event.data)
      // 通知所有已注册的监听器
      listeners.forEach((handler) => {
        try {
          handler(data)
        } catch (err) {
          console.error('[WS] 消息处理错误:', err)
        }
      })
    } catch {
      // 非 JSON 消息（如 ping/pong），忽略
    }
  }

  ws.onclose = (event: CloseEvent) => {
    isConnected.value = false
    console.log(`[WS] 连接关闭 (code: ${event.code})`)

    // 非正常关闭时自动重连
    if (event.code !== 1000 && reconnectAttempts < MAX_RECONNECT_ATTEMPTS) {
      reconnectAttempts++
      const delay = RECONNECT_DELAY_MS * Math.min(reconnectAttempts, 5)
      console.log(`[WS] ${delay}ms 后第 ${reconnectAttempts} 次重连...`)
      reconnectTimer = setTimeout(() => {
        if (token) connectWs(token)
      }, delay)
    }
  }

  ws.onerror = (event: Event) => {
    console.error('[WS] 连接错误:', event)
  }
}

/**
 * 发送消息到服务端（fire-and-forget 模式）
 * @param action - 操作类型，如 "send"、"stop"、"regenerate"
 * @param payload - 请求参数
 */
export function sendWs(action: string, payload: Record<string, any> = {}): void {
  if (!ws || ws.readyState !== WebSocket.OPEN) {
    console.warn('[WS] 连接未就绪，无法发送消息')
    return
  }
  ws.send(JSON.stringify({ action, ...payload }))
}

/**
 * 注册消息监听器
 * @param handler - 收到消息时的回调函数
 * @returns 取消注册的函数
 */
export function onWsMessage(handler: (data: any) => void): () => void {
  listeners.add(handler)
  // 返回取消注册函数
  return () => {
    listeners.delete(handler)
  }
}

/**
 * 断开 WebSocket 连接
 */
export function disconnectWs(): void {
  if (reconnectTimer) {
    clearTimeout(reconnectTimer)
    reconnectTimer = null
  }
  reconnectAttempts = MAX_RECONNECT_ATTEMPTS  // 阻止重连
  if (ws) {
    ws.close(1000, '客户端主动断开')
    ws = null
  }
  isConnected.value = false
}
```

// client/src/stores/chatStore.ts
// 聊天状态管理 —— 会话列表 + 消息 + 流式输出
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { sendWs, onWsMessage, connectWs } from '@/composables/useWs'

// 消息类型定义
export interface Message {
  id: string
  role: 'user' | 'assistant'
  content: string
  model?: string
  token_count?: number
  created_at?: string
}

// 会话类型定义
export interface Conversation {
  id: string
  title: string
  model: string
  tags: string[]
  is_pinned: boolean
  is_starred: boolean
  status: string
  summary: string | null
  total_tokens: number
  message_count: number
  created_at: string
  updated_at: string
}

export const useChatStore = defineStore('chat', () => {
  // ===== 状态 =====
  const conversations = ref<Conversation[]>([])
  const activeId = ref<string | null>(null)           // 当前活跃会话 ID
  const messages = ref<Message[]>([])                  // 当前会话的消息列表
  const isStreaming = ref(false)                       // 是否正在流式输出
  const currentModel = ref('deepseek')                 // 当前使用的模型
  const streamingContent = ref('')                     // 流式输出中的内容缓存
  const streamingMessageId = ref<string | null>(null)  // 流式输出中的消息 ID

  // 取消 WebSocket 监听器的函数
  let wsUnsubscribe: (() => void) | null = null

  // ===== 计算属性 =====
  const activeConversation = computed(() => {
    return conversations.value.find((c) => c.id === activeId.value) || null
  })

  // ===== 初始化 WebSocket 消息监听 =====
  function initWsListener(token: string): void {
    // 先连接
    connectWs(token)

    // 注册消息监听器
    wsUnsubscribe = onWsMessage((data: any) => {
      handleWsMessage(data)
    })
  }

  // 处理 WebSocket 推送的消息
  function handleWsMessage(data: any): void {
    switch (data.type) {
      case 'start':
        // AI 开始生成回复
        isStreaming.value = true
        streamingContent.value = ''
        streamingMessageId.value = data.message_id
        break

      case 'chunk':
        // 收到文本块 —— 追加到流式内容
        if (data.content) {
          streamingContent.value += data.content
        }
        break

      case 'end':
        // 生成完毕 —— 将流式内容写入消息列表
        if (streamingMessageId.value) {
          messages.value.push({
            id: streamingMessageId.value,
            role: 'assistant',
            content: streamingContent.value,
            model: data.model || currentModel.value,
            token_count: data.token_count,
            created_at: new Date().toISOString(),
          })
        }
        isStreaming.value = false
        streamingContent.value = ''
        streamingMessageId.value = null
        break

      case 'error':
        // AI 调用出错
        console.error('[Chat]', data.message)
        isStreaming.value = false
        streamingContent.value = ''
        streamingMessageId.value = null
        break

      case 'stopped':
        // 用户手动停止 —— 保留已生成的内容
        if (streamingMessageId.value && streamingContent.value) {
          messages.value.push({
            id: streamingMessageId.value,
            role: 'assistant',
            content: streamingContent.value + ' [已停止]',
            model: currentModel.value,
            created_at: new Date().toISOString(),
          })
        }
        isStreaming.value = false
        streamingContent.value = ''
        streamingMessageId.value = null
        break
    }
  }

  // ===== 会话列表操作 =====

  /**
   * 从后端获取会话列表
   */
  async function fetchConversations(token: string): Promise<void> {
    try {
      const res = await fetch('/api/conversations?status=active&page=1', {
        headers: { Authorization: `Bearer ${token}` },
      })
      if (res.ok) {
        const data = await res.json()
        conversations.value = data.items
      }
    } catch (err) {
      console.error('[ChatStore] 获取会话列表失败:', err)
    }
  }

  /**
   * 创建新会话
   */
  async function createConversation(token: string, title?: string): Promise<Conversation | null> {
    try {
      const res = await fetch('/api/conversations', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({
          title: title || '新对话',
          model: currentModel.value,
        }),
      })
      if (res.ok) {
        const conv = await res.json()
        conversations.value.unshift(conv)
        return conv
      }
    } catch (err) {
      console.error('[ChatStore] 创建会话失败:', err)
    }
    return null
  }

  /**
   * 删除会话（软删除）
   */
  async function deleteConversation(token: string, id: string): Promise<void> {
    try {
      await fetch(`/api/conversations/${id}`, {
        method: 'DELETE',
        headers: { Authorization: `Bearer ${token}` },
      })
      // 从本地列表移除
      conversations.value = conversations.value.filter((c) => c.id !== id)
      if (activeId.value === id) {
        activeId.value = null
        messages.value = []
      }
    } catch (err) {
      console.error('[ChatStore] 删除会话失败:', err)
    }
  }

  /**
   * 选中会话，加载消息
   */
  async function selectConversation(token: string, id: string): Promise<void> {
    activeId.value = id
    try {
      const res = await fetch(`/api/conversations/${id}`, {
        headers: { Authorization: `Bearer ${token}` },
      })
      if (res.ok) {
        const conv = await res.json()
        messages.value = conv.messages || []
        if (conv.model) {
          currentModel.value = conv.model
        }
      }
    } catch (err) {
      console.error('[ChatStore] 加载会话详情失败:', err)
    }
  }

  // ===== 消息操作 =====

  /**
   * 发送消息（通过 WebSocket）
   */
  function sendMessage(content: string, model?: string): void {
    if (!activeId.value) return

    const msgModel = model || currentModel.value
    const now = new Date().toISOString()

    // 1. 乐观更新 —— 先添加用户消息到列表
    const userMsg: Message = {
      id: `local_${Date.now()}`,
      role: 'user',
      content,
      created_at: now,
    }
    messages.value.push(userMsg)

    // 2. 通过 WebSocket 发送给后端
    sendWs('send', {
      conversation_id: activeId.value,
      content,
      model: msgModel,
    })
  }

  /**
   * 停止当前生成
   */
  function stopGeneration(): void {
    if (!activeId.value || !isStreaming.value) return
    sendWs('stop', { conversation_id: activeId.value })
  }

  /**
   * 重新生成最后一条 AI 回复
   */
  function regenerateLast(): void {
    if (!activeId.value) return
    sendWs('regenerate', { conversation_id: activeId.value })
  }

  /**
   * 切换模型
   */
  function setModel(model: string): void {
    currentModel.value = model
  }

  /**
   * 清理 WebSocket 监听器（组件卸载时调用）
   */
  function cleanup(): void {
    if (wsUnsubscribe) {
      wsUnsubscribe()
      wsUnsubscribe = null
    }
  }

  return {
    // 状态
    conversations,
    activeId,
    messages,
    isStreaming,
    currentModel,
    streamingContent,
    streamingMessageId,
    // 计算属性
    activeConversation,
    // 方法
    initWsListener,
    fetchConversations,
    createConversation,
    deleteConversation,
    selectConversation,
    sendMessage,
    stopGeneration,
    regenerateLast,
    setModel,
    cleanup,
  }
})
```

  return chatStore.conversations.filter(

  )

// 格式化时间 —— 显示相对时间或日期
function formatTime(dateStr: string): string {
  if (!dateStr) return ''
  const date = new Date(dateStr)
  const now = new Date()
  const diffMs = now.getTime() - date.getTime()
  const diffMin = Math.floor(diffMs / 60000)

  if (diffMin < 1) return '刚刚'
  if (diffMin < 60) return `${diffMin} 分钟前`
  const diffHours = Math.floor(diffMin / 60)
  if (diffHours < 24) return `${diffHours} 小时前`
  const diffDays = Math.floor(diffHours / 24)
  if (diffDays < 7) return `${diffDays} 天前`
  return dateStr.slice(0, 10)  // 返回日期部分
}

/**
 * 选中会话
 */
function handleSelect(id: string): void {
  const token = appStore.token
  if (!token) return
  chatStore.selectConversation(token, id)
}

/**
 * 新建会话
 */
async function handleNew(): Promise<void> {
  const token = appStore.token
  if (!token) return
  await chatStore.createConversation(token)
}

/**
 * 右键菜单 —— 显示
 */

  event.preventDefault()
  contextMenu.value = {
    visible: true,
    x: event.clientX,
    y: event.clientY,
    convId,
  }
}

/**
 * 右键菜单 —— 隐藏
 */
function closeContextMenu(): void {
  contextMenu.value.visible = false
}

/**
 * 重命名会话
 */
async function handleRename(): Promise<void> {
  const id = contextMenu.value.convId
  const newTitle = prompt('请输入新标题')
  if (newTitle && newTitle.trim()) {
    const token = appStore.token
    if (!token) return
    try {
      await fetch(`/api/conversations/${id}`, {
        method: 'PUT',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({ title: newTitle.trim() }),
      })
      // 更新本地列表
      const conv = chatStore.conversations.find((c) => c.id === id)
      if (conv) conv.title = newTitle.trim()
    } catch (err) {
      console.error('[ConvList] 重命名失败:', err)
    }
  }
  closeContextMenu()
}

/**
 * 删除会话
 */
async function handleDelete(): Promise<void> {
  const id = contextMenu.value.convId
  const token = appStore.token
  if (!token || !confirm('确定删除这个会话？')) {
    closeContextMenu()
    return
  }
  await chatStore.deleteConversation(token, id)
  closeContextMenu()
}

<script setup lang="ts">

import { marked } from 'marked'
import hljs from 'highlight.js'

// Props
const props = defineProps<{
  message: {
    id: string
    role: 'user' | 'assistant'
    content: string
    model?: string
    token_count?: number
  }
  isStreaming?: boolean  // 是否正在流式输出中

// 配置 marked + highlight.js
marked.setOptions({
  highlight: (code: string, lang: string) => {
    if (lang && hljs.getLanguage(lang)) {
      return hljs.highlight(code, { language: lang }).value
    }
    return hljs.highlightAuto(code).value
  },
})

// 渲染 Markdown 为 HTML
const renderedHtml = computed(() => {
  try {
    return marked.parse(props.message.content) as string
  } catch {
    return props.message.content
  }
})

// 是否为 AI 消息
const isAssistant = computed(() => props.message.role === 'assistant')

 * 复制消息内容到剪贴板

  try {
    await navigator.clipboard.writeText(props.message.content)
    // 简单的复制成功提示（可以用 toast，这里先保持简洁）
  } catch {
    console.error('[ChatMessage] 复制失败')
  }
}
</script>
